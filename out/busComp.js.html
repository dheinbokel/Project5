const PROMPT = require('readline-sync');

let continueUsing, numPassengers, numZones, ticketPrice;
let ticketFairs;

/**
 *   @method
 *   @desc dispatch method for this project
 *   @returns (null)
 */
function main() {

    setPriceArray();
    setContinueUsing();
    while(continueUsing === 1){

        setNumPassengers();
        setNumZones();
        calcTicketPrice();
        displayPrice();

        setContinueUsing();
    }
}
